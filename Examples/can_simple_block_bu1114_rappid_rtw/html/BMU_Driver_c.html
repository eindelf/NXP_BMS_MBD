101static void SPI_Send_Message(uint8_t *data , uint8_t len)



102{



103  DSPI_PUSHR_32B_tag pushr;



104  uint8_t i;



105  pushr.R = 0U;



106



107  // Keep chip select enabled



108  for (i = 0; i < len-1; i++) {



109    pushr.B.TXDATA = data[i];



110    pushr.R |= DSPI_PUSHR_CS6_BITMASK;



111    pushr.R |= DSPI_PUSHR_CONT_EN_BITMASK;



112    DSPI[0]->PUSHR.R = pushr.R;



113  }



114



115  // Disable chip select after last data byte



116  pushr.B.TXDATA = data[i];



117  pushr.R &= DSPI_PUSHR_CONT_DIS_BITMASK;



118  DSPI[0]->PUSHR.R = pushr.R;



119



120  // Indicate end of queue



121  DSPI[0]->SR.R = DSPI_ISR_TCF_RE | DSPI_ISR_EOQF_RE;



122}

263/******************************************************************************

265 *



266 * Inputs:



267 *

269 *

271 *****************************************************************************/



272static void Command_LTC(uint8_t tx_cmd[2])



273{



274  uint8_t cmd[4];



275  uint16_t cmd_pec;



276  cmd[0] = tx_cmd[0];



277  cmd[1] = tx_cmd[1];



278  cmd_pec = Calculate_PEC15(cmd, CMD_LEN);



279  cmd[2] = (uint8_t)(cmd_pec >> 8);



280  cmd[3] = (uint8_t)(cmd_pec);



281  SPI_Send_Message(cmd, 4);



282}

284/******************************************************************************



285 * Function name:   Calculate_PEC15



286 *



287 * Inputs:          uint8_t *data - pointer to data



288 *                  uint8_t len - number of data bytes



289 *



290 * Outputs:         16 bit PEC



291 *



292 * Description:     Calculate PEC based on input data



293 *****************************************************************************/



294static uint16_t Calculate_PEC15(uint8_t *data, uint8_t len)



295{



296  uint16_t rem, addr;



297  uint8_t i;



298



299  // PEC Seed



300  rem = 16;



301  for (i = 0; i < len; i++) {



302    addr = ((rem >> 7) ^ data[i]) & 0xFF;//calculate PEC table address



303    rem = (rem << 8 ) ^ PEC15Table[addr];



304    rem &= 0xFFFF;



305  }



306



307  return ((rem << 1) & 0xFFFF);



308}

320    rem = (rem << 8 ) ^ PEC15Table[addr];

323



324  res[0] = ((rem << 1) & 0xFFFF);
